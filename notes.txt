string newFirstName;

do
{
    Console.Write("Enter new First Name: ");
    newFirstName = Console.ReadLine().Trim(); // Trim the input to remove extra spaces

    if (string.IsNullOrEmpty(newFirstName))
    {
        Console.WriteLine("First Name is required.");
    }

} while (string.IsNullOrEmpty(newFirstName)); // Check if the trimmed string is empty
